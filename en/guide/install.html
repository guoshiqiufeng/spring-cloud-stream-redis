Install

spring-cloud-stream-redis Based on JDK21, support SpringBoot 3.x.

SpringBoot 3.x

spring-cloud-starter-stream-redis

支持Redis 5.x以上

<dependency>
    <groupId>io.github.guoshiqiufeng.cloud</groupId>
    <artifactId>spring-cloud-starter-stream-redis</artifactId>
    <version>0.2.0</version>
</dependency>

implementation 'io.github.guoshiqiufeng.cloud:spring-cloud-starter-stream-redis:0.2.0'

implementation group: 'io.github.guoshiqiufeng.cloud', name: 'spring-cloud-starter-stream-redis', version: '0.2.0'

Bom

<dependencyManagement>
   <dependencies>
       <dependency>
            <groupId>io.github.guoshiqiufeng.cloud</groupId>
            <artifactId>spring-cloud-stream-dependencies</artifactId>
            <version>0.2.0</version>
            <type>pom</type>
            <scope>import</scope>
       </dependency>
   </dependencies>
</dependencyManagement>

dependencies {
    implementation platform("io.github.guoshiqiufeng.cloud:spring-cloud-stream-dependencies:0.2.0")
}
